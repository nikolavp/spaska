28                  .toString(), inputData.getAttributes());
29          clusteredData.addAttribute(new Attribute("cluster",
30                  Attribute.ValueType.Nominal));
31          clusteredData.setClassIndex(clusteredData.getAttributes().size() - 1);
32  
33          for (Instance e : inputData.getElements()) {
34              if (Thread.interrupted()) {
35                  return;
36              }
37  
38              List<Value> list = e.getVector();
39              list.add(new NominalValue("0"));
40              Instance element = new Instance(list);
41              clusteredData.addElement(element);
42          }
43  
44          executionTime = System.currentTimeMillis() - start;
45      }
