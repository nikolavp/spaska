46      /**
47       * Set the class names for a numeric type statistic.
48       * 
49       * @param classNames
50       *            the class names for a numeric type statistic
51       */
52      public void setClassNames(String[] classNames) {
53          if (classNames == null) {
54              throw new IllegalArgumentException("classNames = null!");
55          }
56          if (this.classNames != null) {
57              throw new IllegalStateException("Class names already set!");
58          }
59          if (isNumeric) {
60              throw new IllegalStateException(
61                      "Cannot set class names on numeric type statistic!");
62          }
63          confusionMatrix = new int[classNames.length][classNames.length];
64          namesMap = new HashMap<String, Integer>();
65          this.classNames = new String[classNames.length];
66          for (int i = 0; i < this.classNames.length; i++) {
67              this.classNames[i] = classNames[i];
68              namesMap.put(this.classNames[i], i);
69          }
70          setModified(true);
71      }

73      /*
74       * add a pair (real class, classified class) by class index in confusion
75       * matrix
76       */
77      private void addInfo(int realClass, int classifiedClass) {
78          if (realClass < confusionMatrix.length && realClass >= 0
79                  && classifiedClass < confusionMatrix.length
80                  && classifiedClass >= 0) {
81              confusionMatrix[realClass][classifiedClass]++;
82              setInstances(getInstances() + 1);
83              setModified(true);
84          } else {
85              throw new IllegalArgumentException();
86          }
87      }

98          if (isNumeric) {
99              throw new IllegalStateException(
100                     "Cannot add info by class name on numeric statistics");
101         }
102         if (classNames == null) {

107         addInfo(realClassNumber, classifiedClassNumber);

124         setInstances(getInstances() + 1);

128     /* convert class name to class number */
129     private int getClassNumber(String className) {
130         if (namesMap != null) {
131             Integer classNum = namesMap.get(className);
132             if (classNum != null) {
133                 return classNum;

135         }

147     /* get the mean squared error for a numeric statistic */
148     private double getNumericError() {
149         double sum = 0.0;
150         for (double err : residuals) {
151             sum += err * err;
152         }
153         return (sum / residuals.size());
154     }

156     /* get precision value by class number */
157     private double getPrecision(int classNum) {
158         int returned = 0;
159         for (int i = 0; i < confusionMatrix.length; ++i) {
160             returned += confusionMatrix[i][classNum];
161         }
162         if (returned == 0) {
163             return 0;
164         }
165         return confusionMatrix[classNum][classNum] / (double) returned;
166     }

184     /* get recall value by class number */
185     private double getRecall(int classNum) {
186         int correct = 0;
187         for (int i = 0; i < confusionMatrix.length; ++i) {
188             correct += confusionMatrix[classNum][i];
189         }
190         if (correct == 0) {
191             return 0;
192         }
193         return confusionMatrix[classNum][classNum] / (double) correct;
194     }

246     /**
247      * Return an f-measure by a class name.
248      * 
249      * @param className
250      *            the className for which to get the f-measure metric
251      * @return the f-measure metric for the given class
252      */
253     public double getFMeasure(String className) {
254         if (isNumeric) {
255             throw new UnsupportedOperationException("f-measure not available"
256                     + "for numeric statistics.");
257         }
258         int classNum = getClassNumber(className);
259         double precision = getPrecision(classNum);
260         double recall = getRecall(classNum);
261         if (precision + recall == 0) {
262             return 0;
263         }
264         return (2 * precision * recall) / (precision + recall);
265     }

267     /**
268      * Get an array of all f-measures from the classifier evaluation run.
269      * 
270      * @return an array of all f-measures from the classifier evaluation run
271      */
272     public double[] getFMeasures() {
273         if (isNumeric) {
274             throw new UnsupportedOperationException("f-measure not available"
275                     + "for numeric statistics.");
276         }
277         double[] result = new double[confusionMatrix.length];
278         for (int i = 0; i < result.length; i++) {
279             double p = getPrecision(i);
280             double r = getRecall(i);
281             if (p + r == 0) {
282                 result[i] = 0;
283             } else {
284                 result[i] = (2 * p * r) / (p + r);
285             }
286         }
287         return result;
288     }

335             result.append("Test time (HH:MM:SS.MS): "

340         }

344             correctlyClassified += confusionMatrix[i][i];
345         }
346         result.append("Correctly classified: " + correctlyClassified);
347         double percent = (getInstances() == 0) ? PERCENT_BASE
348                 : (correctlyClassified * PERCENT_BASE)
349                         / (double) getInstances();
350         result.append(String.format(" (%.2f%%)\n", percent));
351         result.append("Incorrectly classified: "
352                 + (getInstances() - correctlyClassified));
353         result.append(String.format(" (%.2f%%)\n", PERCENT_BASE - percent));
354         result.append("Test time (HH:MM:SS.MS): " + timeToString(getTestTime()));
355         result.append("\n================================================");
356         result.append("\nDETAILS BY CLASSES:\n");
357         result.append("------------------------------------------------");
358 
359         for (int i = 0; i < confusionMatrix.length; i++) {
360             result.append("\nClass: " + getClassName(i));
361             int total = 0;
362             for (int j = 0; j < confusionMatrix.length; j++) {
363                 total += confusionMatrix[i][j];
364             }
365             result.append("\nTotal instances: " + total);
366             result.append("\nCorrectly classified: " + confusionMatrix[i][i]
367                     + "\n");
368             percent = (total == 0) ? PERCENT_BASE
369                     : (confusionMatrix[i][i] * PERCENT_BASE) / (double) total;
370             result.append("Incorrectly classified: "
371                     + (total - confusionMatrix[i][i]) + "\n");
372             double precision = getPrecision(i);
373             double recall = getRecall(i);
374             result.append(String.format("Precision:  %.3f\n", precision));
375             result.append(String.format("Recall:     %.3f\n", recall));
376             double fMeasure;

379             } else {
380                 fMeasure = (2 * precision * recall) / (precision + recall);
381             }
382             result.append(String.format("F-Measure:  %.3f\n", fMeasure));
383             result.append("------------------------------------------------");
384         }
385         result.append("\n================================================");
386         setInfo(result.toString());
387         setModified(false);
