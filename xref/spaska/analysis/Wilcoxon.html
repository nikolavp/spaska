82              fstSample[i] = cs1.getGeneralPrecision();
83              fstTime[i] = cs1.getTestTime();
84  
85              CrossValidation cv2 = new CrossValidation();
86              cv2.setFolds(folds);
87              cv2.setSeed(SEEDS.get(i));
88              cv2.setClassifier(classifier2);
89              cv2.setData(dataSet);

92              sndSample[i] = cs2.getGeneralPrecision();
93              sndTime[i] = cs2.getTestTime();

97          statistic.setSecondSamplePopulation(sndSample);

146     @Override
147     public boolean shouldRejectNull(double[] fstSample, double[] sndSample) {
148 
149         WilcoxonTest wTest = null;
150 
151         if (testType == TestType.OneSidedGreaterThan) {
152             wTest = new WilcoxonTest(new PairedData(fstSample, sndSample),
153                     H1.GREATER_THAN, true);
154         } else {
155             if (testType == TestType.OneSidedLessThan) {
156                 wTest = new WilcoxonTest(new PairedData(fstSample, sndSample),
157                         H1.LESS_THAN, true);
158             } else {
159                 wTest = new WilcoxonTest(new PairedData(fstSample, sndSample),
160                         H1.NOT_EQUAL, true);
161             }
162         }
163 
164         double probability = wTest.approxSP() * TEN;
165 
166         boolean shouldReject = false;
167 
168         if (testType == TestType.TwoSided) { // (!=) case
169             if (alpha / 2 < probability) {
170                 shouldReject = true;
171             }
172         } else {
173 
174             if (alpha < probability) {
175                 shouldReject = true;
176             }
177         }
178         return shouldReject;
179 
180     }
