26      @Override
27      public void buildClassifier(Dataset instances) {
28          if (commonValue == null) {
29              int classIndex = instances.getClassIndex();
30              Map<Value, Integer> frequency = new HashMap<Value, Integer>();
31              for (Instance instance : instances.getElements()) {
32                  Value classValue = getAttributeValue(instance, classIndex);
33                  if (frequency.get(classValue) == null) {
34                      frequency.put(classValue, 1);
35                  } else {
36                      frequency.put(classValue, frequency.get(classValue) + 1);
37                  }
38              }
39              int max = 0;
40              for (Map.Entry<Value, Integer> entry : frequency.entrySet()) {
41                  if (max < entry.getValue()) {
42                      commonValue = entry.getKey();
43                      max = entry.getValue();
44                  }
45              }
46          }
47  
48      }
